public class Day2ReverseandPalindrome {
public static void main(String[] args)
{
	int num=256, rev=0;
	int num2=num;
	while(num>0)
	{
		rev=(rev*10)+(num%10);
		num=num/10;
	}
	System.out.println("The reverse is "+rev);
	if(num2==rev)
	{
		System.out.println("PALINDROME");
	}
	else
	{
		System.out.println("NOT A PALINDROME");
	}
}
}
